    // Utility to generate a random integer between min and max (inclusive)
    function randInt(min, max){
      return Math.floor(Math.random() * (max - min + 1)) + min;
    }

    // Format price in EUR with comma separators
    function formatEUR(n){
      return new Intl.NumberFormat('en-EN', { style: 'currency', currency: 'EUR', maximumFractionDigits: 0 }).format(n);
    }

    // Inject rooms into DOM and assign random prices in range 2000-5000
    const roomsGrid = document.getElementById('roomsGrid');
    const priceRange = { min:2000, max:5000 };

    // Shuffle sample images a bit to give visual variety (optional)
    function buildRooms(){
      roomsGrid.innerHTML = '';
      rooms.forEach(room => {
        const price = randInt(priceRange.min, priceRange.max);
        const card = document.createElement('article');
        card.className = 'room-card';
        card.innerHTML = `
          <div class="room-media">
            <img src="${room.img}" alt="${room.name}">

    function openBooking(id, name, price){

      modalBackdrop.setAttribute('aria-hidden','false');
      modalRoomName.textContent = name;
      modalPrice.textContent = formatEUR(price);
      document.getElementById('guestName').focus();
    }

    function closeModal(){
      modalBackdrop.style.display = 'none';
      modalBackdrop.setAttribute('aria-hidden','true');
      document.getElementById('bookingForm').reset();
    }

    // Example simple details

    }

    function confirmBooking(){
      const name = document.getElementById('guestName').value.trim();
      const inDate = document.getElementById('checkIn').value;
      const outDate = document.getElementById('checkOut').value;
      if(!name || !inDate || !outDate){
        alert('Please complete all fields.');
        return;
      }
      // For demo: show a toast via alert (replace with real booking integration)
      alert(`Thanks, ${name}! Your request has been received. We will confirm availability and send an email shortly.`);
      closeModal();
    }
